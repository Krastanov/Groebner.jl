% https://tex.stackexchange.com/questions/432138/automatically-positioning-node-shape-on-to-path-in-tikz-for-outlined-transpa/432147#432147
\tikzset{
  my fat arrow/.style args={width #1 line width #2}{
    -{Triangle[length=#1,width={2*#1}]},line width=#1,, % outer arrow
    postaction={draw,-{Triangle[length={#1-2*#2},width={2*#1-3*sqrt(2)*#2}]},white,
    line width={#1-2*#2},shorten <=#2,shorten >=#2,opacity=1}, % second arrow
  }
}
